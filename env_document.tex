function include_markdown_files(directory, template)
	local lfs = require("lfs")
	for file in lfs.dir(directory) do
		if file:match("%.md$") then
			context("\\processmarkdownfile[template=" .. template .."]{" .. directory .. "/" .. file .."}")
		end
	end
end
