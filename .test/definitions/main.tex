\defineframedtext
  [frameddefinition]
  [
	width=\dimexpr\textwidth + 1cm\relax,
    frame=on,
	rightframe=off,
    framecolor=darkblue,
    rulethickness=2pt,
	location=right,
	offset=0.5cm,
	frameoffset=1cm,
	% align=middle,
    % offset=2ex,
	% loffset=0.5cm,
	% loffset=0.5cm,
    background=color,
    backgroundcolor=lightblue,
  ]


\definehead[definition][subsubsubsubsection][
	number=no,
	style=\bf,
	beforesection={\blank[medium]\offset[leftoffset=-0.5cm]\startframeddefinition},
	aftersection=\stopframeddefinition\blank[medium]
	]
