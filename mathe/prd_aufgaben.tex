\startluacode
local lfs = require("lfs")
for file in lfs.dir("mathe/aufgaben/") do
	if file:match("%.md$") then
		context("\\startmode[" .. file .. "]")
		context("\\gdef\\conditionmacro{0}")
		context("\\processmarkdownfile[template=./.pandoc_templates/aufgaben.context]{mathe/aufgaben/" .. file .. "}")
		context("\\stopmode")

		context("\\doifundefined{conditionmacro}{")

		context("\\executesystemcommand{context --mode=" .. file .. " --result=mathe_aufgaben_" .. string.sub(file, 1, -4) .. " mathe/\\jobname.tex }")

		context("}")
	end
end
\stopluacode
