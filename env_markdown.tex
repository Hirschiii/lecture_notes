\startenvironment env_markdown

\usemodule[filter]

\defineexternalfilter
  [markdown]
  [filtercommand={pandoc -t context 
				  --template=\externalfilterparameter{template} 
                  -i \externalfilterinputfile\space
                  -o \externalfilteroutputfile},
   format=markdown,
   cache=no,
   directory=.output_pandoc]

\stopenvironment
